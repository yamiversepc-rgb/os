#include <stdio.h>
#include <fcntl.h>
#include <unistd.h>

int main() {
    int fd;
    char buffer[100];

    fd = open("sample.txt", O_CREAT | O_RDWR, 0777);

    if (fd < 0) {
        printf("Error opening file\n");
        return 1;
    }

    write(fd, "Hello OS Lab\n", 13);

    lseek(fd, 0, SEEK_SET);
    read(fd, buffer, 13);

    printf("File Contents: %s\n", buffer);

    close(fd);
    return 0;
}
