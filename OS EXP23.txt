#include <stdio.h>

int main() {
    int b, p;
    printf("Enter number of blocks: ");
    scanf("%d", &b);

    int block[b];
    printf("Enter block sizes:\n");
    for (int i = 0; i < b; i++)
        scanf("%d", &block[i]);

    printf("Enter number of processes: ");
    scanf("%d", &p);

    int process[p];
    printf("Enter process sizes:\n");
    for (int i = 0; i < p; i++)
        scanf("%d", &process[i]);

    for (int i = 0; i < p; i++) {
        int allocated = 0;
        for (int j = 0; j < b; j++) {
            if (block[j] >= process[i]) {
                printf("Process %d allocated to Block %d\n", process[i], block[j]);
                block[j] -= process[i];
                allocated = 1;
                break;
            }
        }
        if (!allocated)
            printf("Process %d not allocated\n", process[i]);
    }

    return 0;
}
